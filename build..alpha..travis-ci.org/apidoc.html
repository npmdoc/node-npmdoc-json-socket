function json-socket.sendSingleMessageAndReceive (port, host, message, callback)

  • description and source-code
    sendSingleMessageAndReceive = function (port, host, message, callback) {
    callback = callback || function(){};
    var socket = new JsonSocket(new net.Socket());
    socket.connect(port, host);
    socket.on('error', function(err) {
        callback(err);
    });
    socket.on('connect', function() {
        socket.sendMessage(message, function(err) {
            if (err) {
                socket.end();
                return callback(err);
            }
            socket.on('message', function(message) {
                socket.end();
                if (message.success === false) {
                    return callback(new Error(message.message));
                }
                callback(null, message)
            });
        });
    });
}
